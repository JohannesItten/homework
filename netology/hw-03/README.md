### Ответ 9
Создавал виртуальные машины терраформом в проксмоксе, заббикс ставил через cloud-init
```yaml
#cloud-config

package_update: true
package_upgrade: false

packages:
  - wget

runcmd:
  - wget https://repo.zabbix.com/zabbix/7.4/release/ubuntu/pool/main/z/zabbix-release/zabbix-release_latest_7.4+ubuntu24.04_all.deb 
  - dpkg -i zabbix-release_latest_7.4+ubuntu24.04_all.deb
  - apt update

  - apt install -y zabbix-agent2

  - sed -i 's/^Server=.*/Server=10.5.1.213/' /etc/zabbix/zabbix_agent2.conf
  - sed -i "s/^Hostname=.*/Hostname=$(hostname)/" /etc/zabbix/zabbix_agent2.conf

  - systemctl enable zabbix-agent2
  - systemctl restart zabbix-agent2
```
